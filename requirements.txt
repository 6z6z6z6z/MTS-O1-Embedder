#       --index-url https://download.pytorch.org/whl/cu128

# ── Step 2: Triton 3.3.1 ──────────────────────────────────────

#
#   pip uninstall triton pytorch-triton triton-nightly -y
#   pip install triton==3.3.1
#
# ── Step 3: flash-linear-attention 0.4.2 ─────────────────────
# Qwen3.5 GDR 层必须；0.4.2 是与 Triton 3.3.1 对应的稳定版本
# 必须从源码安装（--no-deps 避免覆盖已装版本）
#
#   pip uninstall flash-linear-attention -y
#   pip install einops ninja
#   pip install --no-use-pep517 --no-deps \
#       git+https://github.com/fla-org/flash-linear-attention@v0.4.2
#
# ── Step 4: 本文件其余依赖 ────────────────────────────────────
#
#   pip install -r requirements.txt
#
# ── 验证 ──────────────────────────────────────────────────────
